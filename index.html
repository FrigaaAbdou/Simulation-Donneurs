import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# Simulation et analyse des donnes sur les donneurs de sang

# 1. Gnration des donnes simules
n_donneurs = 5000
np.random.seed(42)

# Donnes dmographiques
ages = np.random.randint(18, 71, n_donneurs)
sexes = np.random.choice(['F', 'M'], size=n_donneurs, p=[0.55, 0.45])
freq_dons = np.random.choice(['unique', 'regulier', 'frequent'], size=n_donneurs, p=[0.3, 0.5, 0.2])

# Donnes biologiques
hb = np.where(sexes == 'M', np.random.normal(14.5, 1.2, n_donneurs), np.random.normal(13.2, 1.0, n_donneurs))
ferritine = np.where(sexes == 'M', np.random.normal(100, 50, n_donneurs), np.random.normal(30, 15, n_donneurs))

# 2. Calcul du statut de ferritine
def calc_statut_ferritine(ferritine, hb, sexe):
    if sexe == 'M' and ferritine < 15:
        return 'carence'
    elif sexe == 'F' and ferritine < 15:
        return 'carence'
    else:
        return 'normal'

statut_ferritine = [calc_statut_ferritine(f, h, s) for f, h, s in zip(ferritine, hb, sexes)]

# Cration du DataFrame
data = pd.DataFrame({
    'Age': ages,
    'Sexe': sexes,
    'Frequence_don': freq_dons,
    'Hb': hb,
    'Ferritine': ferritine,
    'Statut_ferritine': statut_ferritine
})

# Aperu des donnes simules
print("Aperu des donnes simules :")
print(data.head())

# 3. Analyse statistique et graphique
# Rpartition des donneurs par frquence de don et statut de ferritine
carence_par_frequence = data.groupby(['Frequence_don', 'Statut_ferritine']).size().unstack()

# Visualisation
plt.figure(figsize=(10, 6))
carence_par_frequence.plot(kind='bar', stacked=True, figsize=(10, 6), color=['skyblue', 'salmon'])
plt.title('Rpartition des donneurs par frquence de don et statut de ferritine')
plt.ylabel('Nombre de donneurs')
plt.xlabel('Frquence de don')
plt.xticks(rotation=0)
plt.legend(title='Statut Ferritine')
plt.grid(axis='y', linestyle='--', alpha=0.7)

# Affichage du graphique
plt.tight_layout()
plt.show()
